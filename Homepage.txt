import React, { useState, useEffect } from "react";
import { Realm } from "@/entities/Realm";
import { UserPreferences } from "@/entities/UserPreferences";
import { User } from "@/entities/User";
import { Button } from "@/components/ui/button";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Badge } from "@/components/ui/badge";
import { Heart, Users, Sparkles, Clock, Star, Brain, Shield } from "lucide-react";
import { useNavigate } from "react-router-dom";
import { createPageUrl } from "@/utils";

const realmIcons = {
  loneliness: Heart,
  gut_health: Sparkles,
  burnout: Clock,
  digital_detox: Star,
  self_doubt: Brain,
  relationships: Users,
  financial_planning: Star,
  personal_growth: Sparkles,
  spirituality: Star,
  vulnerability: Heart,
  navigating_change: Shield
};

const realmColors = {
  loneliness: 'from-indigo-500 to-purple-600',
  gut_health: 'from-emerald-500 to-teal-600',
  burnout: 'from-amber-500 to-orange-600',
  digital_detox: 'from-cyan-500 to-blue-600',
  self_doubt: 'from-violet-500 to-purple-600',
  relationships: 'from-rose-500 to-pink-600',
  financial_planning: 'from-green-500 to-emerald-600',
  personal_growth: 'from-yellow-500 to-amber-600',
  spirituality: 'from-purple-500 to-indigo-600',
  vulnerability: 'from-pink-500 to-rose-600',
  navigating_change: 'from-blue-500 to-indigo-600'
};

export default function Home() {
  const [realms, setRealms] = useState([]);
  const [userPreferences, setUserPreferences] = useState(null);
  const [isLoading, setIsLoading] = useState(true);
  const navigate = useNavigate();

  useEffect(() => {
    loadData();
  }, []);

  const loadData = async () => {
    try {
      const user = await User.me();
      const [realmData, prefData] = await Promise.all([
        Realm.list(),
        UserPreferences.filter({ created_by: user.email })
      ]);
      
      setRealms(realmData);
      setUserPreferences(prefData[0] || null);
    } catch (error) {
      console.error("Error loading data:", error);
    }
    setIsLoading(false);
  };

  const handleRealmSelect = (realm) => {
    navigate(createPageUrl(`MetaphorFlow?realm=${realm.id}`));
  };

  if (isLoading) {
    return (
      <div className="p-8 max-w-7xl mx-auto">
        <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">
          {Array(6).fill(0).map((_, i) => (
            <div key={i} className="glass-morphism rounded-2xl p-6 animate-pulse">
              <div className="h-48 bg-white/5 rounded-lg mb-4"></div>
              <div className="h-4 bg-white/5 rounded mb-2"></div>
              <div className="h-3 bg-white/5 rounded w-3/4"></div>
            </div>
          ))}
        </div>
      </div>
    );
  }

  return (
    <div className="p-4 md:p-8 max-w-7xl mx-auto">
      {/* Hero Section */}
      <div className="text-center mb-12">
        <div className="glass-morphism rounded-3xl p-8 md:p-12 mb-8">
          <div className="flex justify-center mb-6">
            <div className="w-20 h-20 rounded-3xl bg-gradient-to-br from-blue-500 to-indigo-600 flex items-center justify-center">
              <Brain className="w-10 h-10 text-white" />
            </div>
          </div>
          <h1 className="text-4xl md:text-6xl font-bold text-amber-dark mb-6 leading-tight">
            Welcome to
            <span className="block bg-gradient-to-r from-amber-800 to-amber-900 bg-clip-text text-transparent">
              Mind Shelter
            </span>
          </h1>
          <p className="text-xl text-amber-light mb-8 max-w-2xl mx-auto leading-relaxed">
            A sanctuary for reflection and connection through storytelling. Share your journey, 
            listen deeply to others, and find solace in our shared human experience.
          </p>
          <div className="flex flex-wrap gap-4 justify-center">
            <Badge className="bg-amber-100 text-amber-900 border-amber-200 px-4 py-2">
              <Users className="w-4 h-4 mr-2" />
              Intimate 5-person circles
            </Badge>
            <Badge className="bg-amber-200 text-amber-900 border-amber-300 px-4 py-2">
              <Clock className="w-4 h-4 mr-2" />
              Mindful 15-minute sessions
            </Badge>
            <Badge className="bg-amber-300 text-amber-900 border-amber-400 px-4 py-2">
              <Brain className="w-4 h-4 mr-2" />
              AI-guided matching
            </Badge>
          </div>
        </div>
      </div>

      {/* Realms Grid */}
      <div className="mb-8">
        <h2 className="text-2xl md:text-3xl font-bold text-gray-900 mb-2">Explore Reflective Realms</h2>
        <p className="text-gray-700 text-lg mb-8">Choose the space that resonates with your current journey</p>
        <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">
          {realms.map((realm) => {
            const IconComponent = realmIcons[realm.category] || Sparkles;
            const colorClass = realmColors[realm.category] || 'from-blue-500 to-indigo-600';
            
            return (
              <Card key={realm.id} className="glass-morphism border-white/10 hover:bg-amber-600/90 transition-all duration-300 group cursor-pointer transform hover:scale-102 hover:shadow-xl flex flex-col h-full">
                <CardHeader className="pb-4 flex-shrink-0">
                  <div className={`w-16 h-16 rounded-2xl bg-gradient-to-br ${colorClass} flex items-center justify-center mb-4 group-hover:scale-110 transition-transform duration-300`}>
                    <IconComponent className="w-8 h-8 text-white" />
                  </div>
                  <CardTitle className="text-amber-dark text-xl font-semibold">
                    {realm.name}
                  </CardTitle>
                </CardHeader>
                <CardContent className="flex-grow flex flex-col">
                  <p className="text-amber-light mb-6 leading-relaxed flex-grow">
                    {realm.description}
                  </p>
                  <Button 
                    onClick={() => handleRealmSelect(realm)}
                    className="w-full bg-gradient-to-r from-blue-500 to-indigo-600 hover:from-blue-600 hover:to-indigo-700 text-white border-0 rounded-xl py-3 font-medium transition-all duration-300 mt-auto"
                  >
                    Enter Realm
                    <Sparkles className="w-4 h-4 ml-2" />
                  </Button>
                </CardContent>
              </Card>
            );
          })}
        </div>
      </div>

      {/* Community Insights */}
      <div className="glass-morphism rounded-3xl p-6 md:p-8">
        <h3 className="text-xl font-semibold text-amber-dark mb-6 text-center">Community Insights</h3>
        <div className="grid grid-cols-1 md:grid-cols-3 gap-8">
          <div className="text-center">
            <div className="w-16 h-16 rounded-2xl bg-gradient-to-br from-blue-500 to-indigo-600 flex items-center justify-center mx-auto mb-4">
              <Sparkles className="w-8 h-8 text-white" />
            </div>
            <div className="text-3xl font-bold text-amber-dark mb-2">2,847</div>
            <div className="text-amber-light">Stories Shared</div>
            <div className="text-amber-light/80 text-sm mt-1">Moments of vulnerability and growth</div>
          </div>
          <div className="text-center">
            <div className="w-16 h-16 rounded-2xl bg-gradient-to-br from-indigo-500 to-purple-600 flex items-center justify-center mx-auto mb-4">
              <Users className="w-8 h-8 text-white" />
            </div>
            <div className="text-3xl font-bold text-amber-dark mb-2">1,293</div>
            <div className="text-amber-light">Active Seekers</div>
            <div className="text-amber-light/80 text-sm mt-1">Individuals on their journey of reflection</div>
          </div>
          <div className="text-center">
            <div className="w-16 h-16 rounded-2xl bg-gradient-to-br from-purple-500 to-pink-600 flex items-center justify-center mx-auto mb-4">
              <Heart className="w-8 h-8 text-white" />
            </div>
            <div className="text-3xl font-bold text-amber-dark mb-2">4.9</div>
            <div className="text-amber-light">Connection Rating</div>
            <div className="text-amber-light/80 text-sm mt-1">Average depth of meaningful connection</div>
          </div>
        </div>
      </div>
    </div>
  );
}
